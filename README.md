# sTunes_Custom_SQL_Querying
The following is a list of certain syntax used with SQLite. 

# Clauses 
    GROUP BY (aggregate)
    LIMIT
    ORDER BY
    WHERE (non-aggregate)
# Aggregate Functions
    AVG
    COUNT
    SUM
  # String Functions
    CONCATENATION
    STRFTIME
    UPPER
  # Other Functions
    Round
# Joins
    INNER JOIN

# Comparison Operators
    =
    >=
    <=
# Logical Operators
    AND

# Conventions
    tablename.FieldName
    JOIN ON foreignkey.dimensiontable = primarykey.facttable
    
# Other Syntax
    Alias
    SELECT subquery
