# Clauses 
    GROUP BY (aggregate)
    LIMIT
    ORDER BY
    WHERE (non-aggregate)
# Aggregate Functions
    AVG
    COUNT
    SUM
# String Functions
    CONCATENATION
    STRFTIME
    UPPER
# Other Functions
    ROUND
# Joins
    INNER JOIN
# Comparison Operators
    =
    >=
    <=
# Logical Operators
    AND
# Conventions
    tablename.FieldName
    JOIN ON foreignkey.dimensiontable = primarykey.facttable   
# Other Syntax
    Alias
    SELECT subquery
